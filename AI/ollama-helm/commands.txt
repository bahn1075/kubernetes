# 1. ollama helm 설치
helm repo add otwld https://helm.otwld.com/
helm repo update

 helm upgrade -install ollama otwld/ollama \
  --namespace ollama \
  --create-namespace \
  -f /app/kubernetes/AI/ollama-helm/values.yaml
